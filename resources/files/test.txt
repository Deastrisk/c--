int: choice = 0;
double: num1;
double: num2;

while (true) {
    echo("\n+--- SIMPLE CALCULATOR ---+\n");
    echo("| 1. Addition\n");
    echo("| 2. Subtraction\n");
    echo("| 3. Multiply\n");
    echo("| 4. Division\n");
    echo("| 5. Power\n");
    echo("| 0. Exit\n");
    echo("+-------------------------+\n");
    echo("| choice (0-5): ");
    choice = (int) input();

    if (!choice) {
        break;
    }

    if (choice < 0 || choice > 5) {
        echo("| Invalid choice");
        continue;
    }

    echo("| number 1: ");
    num1 = (double) input();

    echo("| number 2: ");
    num2 = (double) input();

    if (choice == 1) {
        echo("| ADDITION RESULT: {num1 + num2}");
    } else if (choice == 2) {
        echo("| SUBTRACTION RESULT: {num1 - num2}");
    } else if (choice == 3) {
        echo("| MULTIPLY RESULT: {num1 * num2}");
    } else if (choice == 4) {
        echo("| DIVISION RESULT: {num1 / num2}");
    } else if (choice == 5) {
        double: result = 1;
        for (int: i = 0; i < num2; i += 1) {
             result *= num1;
        }
        echo("| POWER RESULT: {result}");
    }
}
